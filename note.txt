
build images

docker build .\accounts\.  -t esmaeeil/accounts:2.0
docker build .\loans\.  -t esmaeeil/loans:2.0
docker build .\cards\.  -t esmaeeil/cards:2.0
docker build .\configserver\  -t esmaeeil/configserver:1.0
docker build .\eurekaserver\  -t esmaeeil/eurekaserver:1.0

docker build .\accounts\.  -t esmaeeil/accounts:2.0; docker build .\loans\.  -t esmaeeil/loans:2.0; docker build .\cards\.  -t esmaeeil/cards:2.0; docker build .\configserver\  -t esmaeeil/configserver:1.0; docker build .\eurekaserver\  -t esmaeeil/eurekaserver:1.0
